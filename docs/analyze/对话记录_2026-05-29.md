```python
def _render_self_trajectory(si):
    lines = []
    
    # 1. 状态变化
    states = si.perception.agent_state_history[-5:]
    if len(states) >= 2 and states[0] != states[-1]:
        lines.append(f"状态变化：{states[0]} → {states[-1]}")
    
    # 2. 目标推进
    prog = si.mind.goal_progress_history[-5:]
    if len(prog) >= 2:
        diff = prog[-1] - prog[0]
        if diff > 0.1:
            lines.append(f"目标在推进（{prog[0]:.0%} → {prog[-1]:.0%}，+{diff:.0%}）")
        elif diff < -0.05:
            lines.append(f"目标有所回退（{prog[0]:.0%} → {prog[-1]:.0%}）")
        else:
            lines.append(f"目标进展平稳（{prog[-1]:.0%}）")
    
    # 3. Drive 差值：取当前值（实时代理）vs 上次渲染时存的快照
    current_energy = si.body.energy
    last_snapshot = si.history._rendered_body_snapshot  # 上次渲染时存的
    if last_snapshot:
        energy_diff = current_energy - last_snapshot.get("energy", current_energy)
        emotion = si.body.emotion_type
        if abs(energy_diff) > 0.1:
            direction = "上升" if energy_diff > 0 else "下降"
            lines.append(f"能量{direction}（{energy_diff:+.0%}），当前心情{emotion}")
    
    if not lines:
        return ""
    lines.insert(0, "【自我轨迹】")
    return "\n".join(lines)
```
